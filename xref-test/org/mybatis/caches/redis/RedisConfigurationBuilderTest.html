141   private String findUnsupportedPropertyName() {
142     for (Method method : RedisConfig.class.getMethods()) {
143       if (!method.getName().startsWith("set") || method.getParameterCount() != 1) {
144         continue;
145       }
146       String propertyName = Introspector.decapitalize(method.getName().substring(3));
147       if ("serializer".equals(propertyName) || "sslSocketFactory".equals(propertyName)
148           || "sslParameters".equals(propertyName) || "hostnameVerifier".equals(propertyName)) {
149         continue;
150       }
151       Class<?> type = method.getParameterTypes()[0];
152       if (!isSupportedPropertyType(type)) {
153         return propertyName;
154       }
155     }
156     return null;
157   }
158 
159   private boolean isSupportedPropertyType(Class<?> type) {
160     return String.class == type || int.class == type || Integer.class == type || long.class == type
161         || Long.class == type || short.class == type || Short.class == type || byte.class == type || Byte.class == type
162         || float.class == type || Float.class == type || boolean.class == type || Boolean.class == type
163         || double.class == type || Double.class == type;
164   }

166   private static class PropertiesClassLoader extends ClassLoader {
167     private final byte[] propertyBytes;
168 
169     private PropertiesClassLoader(String propertiesContent) {
170       propertyBytes = propertiesContent.getBytes(StandardCharsets.UTF_8);
171     }
172 
173     @Override
174     public InputStream getResourceAsStream(String name) {
175       return new ByteArrayInputStream(propertyBytes);
176     }
177   }
178 
179   private static class IOExceptionClassLoader extends ClassLoader {
180     @Override
181     public InputStream getResourceAsStream(String name) {
182       return new InputStream() {
183         @Override
184         public int read() throws IOException {
185           throw new IOException("simulated IO error");
186         }
187       };
188     }
189   }
